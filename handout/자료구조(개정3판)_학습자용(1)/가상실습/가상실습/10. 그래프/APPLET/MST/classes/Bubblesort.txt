void prim(int s)
{
 int i, u, v;

 for(u=0;u<VERTICES;u++)		
   distance[u]=INF;
 distance[s]=0;				
 for(i=0;i<VERTICES;i++){
   u = get_min_vertex();
   selected[u]=TRUE;				
   if( distance[u] == INF ) return;	
   printf("%d ", u);				
   for( v=0; v<VERTICES; v++)
      if( adj_mat[u][v]!= INF)	
          if( !selected[v] && weight[u][v]<distance[v] )
              distance[v] = weight[u][v];
 }
}
