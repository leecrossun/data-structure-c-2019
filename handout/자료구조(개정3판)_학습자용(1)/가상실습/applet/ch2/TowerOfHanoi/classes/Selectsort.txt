interpolation_search(int list[], int n, int key)
{
 int left = 0;
 int right = n-1;
 int middle;
 while( left < right ){       
   middle = left + (key-list[left])*(right-left)/(list[right]-list[left]);
   if( middle < 0 ) middle = 0;
   if( middle >= n) middle = n-1;
   if( key > list[middle]))
        left = middle+1;
   else if( key > list[middle]))
        right = middle-1; 
   else 
        return middle;
 }
 return -1;
}
